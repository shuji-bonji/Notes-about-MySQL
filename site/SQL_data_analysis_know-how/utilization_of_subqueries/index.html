    サブクエリの活用¶

    サブクエリを利用して、複数のクエリを連結して複雑なSQLクエリを作成する方法を学習します。

    FROM句におけるサブクエリ¶

    FROM句におけるサブクエリにはASが必須になる。

    基本型¶

    SELECT
    name
FROM
    (
    SELECT
        *
    FROM
        users
    )
    AS
        sub;

    ->     users.birthday,
    ->     order_details.amount * products.price AS fee,
    ->     order_details.amount * products.price * 1.1 AS tax_included
    -> FROM
    ->     users
    -> INNER JOIN
    ->     orders
    -> ON
    ->     users.id = orders.user_id
    -> INNER JOIN
    ->     order_details
    -> ON
    ->     orders.id = order_details.order_id
    -> INNER JOIN
    ->     products
    -> ON
    ->     order_details.product_id = products.id
    -> GROUP BY
    ->     users.id
    -> HAVING
    ->     SUM(amount * price ) <
    ->     (
    ->     SELECT
    ->         SUM(price)
    -> FROM
    ->         products
    -> )
    -> ORDER BY
    ->     users.id
    -> ;
